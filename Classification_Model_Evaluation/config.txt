def extract_and_sum_weights(layer_profile_data):
    """
    Parses TensorRT layer profile data to find all 'Weights' metadata dictionaries
    and calculates the total sum of the weight counts.
    """
    weights_dicts = []
    total_weights_count = 0

    # Ensure data is an iterable list of layer dictionaries
    if isinstance(layer_profile_data, list):
        for layer in layer_profile_data:
            # Check if layer is a dictionary and contains the 'Weights' key
            if isinstance(layer, dict) and "Weights" in layer:
                weight_info = layer["Weights"]
                weights_dicts.append(weight_info)
                
                # Safely extract and sum the total number of weights (Count)
                if "Count" in weight_info:
                    total_weights_count += weight_info["Count"]
                    
    return weights_dicts, total_weights_count
